  <action>
    Replace static import with dynamic import + preload:
    
    **Current:**
    ```typescript
    import CommandPalette from '@/components/command-palette';
    ```
    
    **Target:**
    ```typescript
    import dynamic from 'next/dynamic';
    import { useCallback, useState } from 'react';
    
    const CommandPalette = dynamic(
      () => import('@/components/command-palette'),
      { ssr: false }
    );
    
    // In component:
    const [showCommandPalette, setShowCommandPalette] = useState(false);
    const [isLoaded, setIsLoaded] = useState(false);
    
    const preloadCommandPalette = useCallback(() => {
      if (!isLoaded) {
        import('@/components/command-palette');
        setIsLoaded(true);
      }
    }, [isLoaded]);
    ```
    
    Add preload trigger to search button:
    ```tsx
    <Button 
      onMouseEnter={preloadCommandPalette}
      onFocus={preloadCommandPalette}
      onClick={() => setShowCommandPalette(true)}
    >
      Search...
    </Button>
    ```
    
    This defers the 373-line component and preloads it on hover/focus.
  </action>
  <verify>

  </verify>
  <done>
    - CommandPalette is dynamically imported
    - Preload function triggers on hover/focus
    - SSR disabled
    - Static import removed
  </done>
