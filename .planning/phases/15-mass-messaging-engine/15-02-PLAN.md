```typescript
/**
 * POST /api/campaigns/[id]/hydrate
 * 
 * Manually hydrate a campaign to the queue.
 * Used for:
 * - Starting a draft campaign
 * - Re-hydrating a paused campaign
 * 
 * Body: { variationOptions?: VariationOptions }
 */
export async function POST(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  // 1. Auth check (requirePermission for campaigns.manage)
  
  // 2. Load campaign
  const campaign = await getCampaign(id);
  if (!campaign) return 404;
  
  // 3. Check status (only draft or paused can be hydrated)
  if (!['draft', 'paused'].includes(campaign.status)) {
    return 400 { error: 'Campaign already in progress or completed' };
  }
  
  // 4. Hydrate
  const result = await hydrateCampaignToQueue(id, body.variationOptions);
  
  // 5. Update campaign status to 'sending'
  await updateCampaign(id, { status: 'sending' });
  
  // 6. Return result
  return { success: true, ...result };
}
```

Update the existing campaign cron to hydrate scheduled campaigns to the queue.

**Current behavior** (from Phase 09): The cron finds scheduled campaigns where scheduledAt has passed and sends them directly via sendText.

**New behavior:**
1. Find scheduled campaigns where scheduledAt has passed
2. For each campaign:
   a. Hydrate to queue (using hydrateCampaignToQueue)
   b. Update status to 'sending'
   c. The send-queue cron will pick up the messages

**This separates concerns:**
- Campaign cron: hydration + status management
- Send-queue cron: actual message delivery

Update `src/app/api/cron/campaigns/route.ts`:

```typescript
export async function GET(request: NextRequest) {
  // Auth check...
  
  const now = new Date();
  
  // Find campaigns scheduled for now or earlier that are still 'scheduled'
  const dueCampaigns = await db
    .select()
    .from(campaigns)
    .where(
      and(
        eq(campaigns.status, 'scheduled'),
        lte(campaigns.scheduledAt, now)
      )
    );
  
  const results = [];
  
  for (const campaign of dueCampaigns) {
    try {
      // Hydrate to queue instead of direct send
      const hydrationResult = await hydrateCampaignToQueue(campaign.id);
      
      // Update status
      await updateCampaign(campaign.id, { status: 'sending' });
      
      results.push({
        campaignId: campaign.id,
        name: campaign.name,
        ...hydrationResult
      });
    } catch (err) {
      results.push({
        campaignId: campaign.id,
        name: campaign.name,
        error: String(err)
      });
    }
  }
  
  return NextResponse.json({
    timestamp: now.toISOString(),
    processed: results.length,
    campaigns: results
  });
}
```
