# Plan 21-01: Schema Enhancement & Segment-Group Mapping

## Objective
Adicionar campos necessários para associar segmentos a grupos e chips, criando o foundation para a integração completa.

## Context
Atualmente, segments, chips e groups existem de forma isolada. Precisamos de uma forma de conectar:
- Segment → Chip (qual chip envia para este segmento)
- Segment → Group (qual grupo recebe leads deste segmento)
- Chip → Segments (quais segmentos este chip atende)

// src/lib/segment-mapping.ts
interface SegmentMapping {
  segmentId: string;
  segmentName: string;
  segmentTag: string;
  chip: {
    id: string;
    name: string;
    healthStatus: string;
    availableCapacity: { daily: number; hourly: number };
  } | null;
  group: {
    id: string;
    name: string;
    inviteUrl: string;
    capacity: { current: number; max: number; percent: number };
    status: string;
  } | null;
}

export async function getSegmentGroupMapping(segmentTag: string): Promise<SegmentMapping | null>;
export async function getAllSegmentMappings(): Promise<SegmentMapping[]>;
