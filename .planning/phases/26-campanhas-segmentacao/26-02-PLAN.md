2. **Compute usage stats per segment**:
   ```typescript
   const segmentUsage = useMemo(() => {
     const usage = new Map<string, { count: number; lastUsed: string | null }>();
     for (const campaign of campaigns) {
       if (!campaign.segmentId) continue;
       const existing = usage.get(campaign.segmentId);
       const campaignDate = campaign.updatedAt;
       if (!existing) {
         usage.set(campaign.segmentId, { count: 1, lastUsed: campaignDate });
       } else {
         existing.count++;
         if (campaignDate && (!existing.lastUsed || campaignDate > existing.lastUsed)) {
           existing.lastUsed = campaignDate;
         }
       }
     }
     return usage;
   }, [campaigns]);
   ```
