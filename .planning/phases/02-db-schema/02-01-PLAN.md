```typescript
export const voters = pgTable('voters', {
  id: uuid('id').primaryKey().default(sql`gen_random_uuid()`),
  name: text('name').notNull(),
  cpf: text('cpf'),                    // Optional — some imports may not have CPF
  phone: text('phone').notNull(),
  zone: text('zone'),                   // Electoral zone
  section: text('section'),             // Electoral section
  city: text('city'),
  neighborhood: text('neighborhood'),   // Bairro
  tags: text('tags').array().default(sql`'{}'`),  // Flexible tagging: ['indeciso', 'apoiador', 'zona-sul']
  engagementScore: integer('engagement_score').default(0),  // 0-100, computed from interactions
  optInStatus: text('opt_in_status', { enum: ['active', 'expired', 'revoked', 'pending'] }).default('pending'),
  optInDate: timestamp('opt_in_date', { withTimezone: true }),
  lastContacted: timestamp('last_contacted', { withTimezone: true }),
  contactCount: integer('contact_count').default(0),
  enabled: boolean('enabled').default(true),
  createdAt: timestamp('created_at', { withTimezone: true }).default(sql`now()`),
  updatedAt: timestamp('updated_at', { withTimezone: true }).default(sql`now()`),
}, (t) => [
  index('idx_voters_phone').on(t.phone),
  index('idx_voters_zone').on(t.zone),
  index('idx_voters_opt_in').on(t.optInStatus),
  index('idx_voters_engagement').on(t.engagementScore),
]);
```

**campaigns table:**
```typescript
export const campaigns = pgTable('campaigns', {
  id: uuid('id').primaryKey().default(sql`gen_random_uuid()`),
  name: text('name').notNull(),
  template: text('template').notNull(),     // Message template with {nome}, {bairro} variables
  variables: text('variables').array().default(sql`'{}'`),  // List of variable names used
  status: text('status', { enum: ['draft', 'scheduled', 'sending', 'sent', 'paused', 'cancelled'] }).default('draft'),
  segmentId: uuid('segment_id').references(() => segments.id, { onDelete: 'set null' }),
  scheduledAt: timestamp('scheduled_at', { withTimezone: true }),
  windowStart: time('window_start').default('08:00'),
  windowEnd: time('window_end').default('22:00'),
  // A/B testing
  abEnabled: boolean('ab_enabled').default(false),
  abVariantB: text('ab_variant_b'),         // Alternative message template
  abSplitPercent: integer('ab_split_percent').default(50),  // % that gets variant A
  // Stats (denormalized for fast dashboard reads)
  totalSent: integer('total_sent').default(0),
  totalDelivered: integer('total_delivered').default(0),
  totalRead: integer('total_read').default(0),
  totalReplied: integer('total_replied').default(0),
  totalFailed: integer('total_failed').default(0),
  createdAt: timestamp('created_at', { withTimezone: true }).default(sql`now()`),
  updatedAt: timestamp('updated_at', { withTimezone: true }).default(sql`now()`),
}, (t) => [
  index('idx_campaigns_status').on(t.status),
  index('idx_campaigns_segment').on(t.segmentId),
]);
```

**src/lib/db-voters.ts:**
Functions:
- `loadVoters()` — select all voters, ordered by createdAt desc
- `getVoter(id: string)` — select single voter by id, return first or undefined
- `addVoter(data: NewVoter)` — insert one voter, return inserted
- `updateVoter(id: string, data: Partial<NewVoter>)` — update by id with set + updatedAt, return updated
- `deleteVoter(id: string)` — delete by id
- `bulkInsertVoters(data: NewVoter[])` — insert many voters in a single query (for CSV import). Use `db.insert(voters).values(data).returning()`. Handle empty array gracefully (return []).
- `searchVoters(query: string)` — search by name or phone using `ilike` pattern `%${query}%`, limit 50
- `getVotersBySegment(segmentId: string)` — join segmentVoters → voters for a given segment

All functions are async, return typed results. Export all functions as named exports.

**src/lib/db-campaigns.ts:**
Functions:
- `loadCampaigns()` — select all, ordered by createdAt desc
- `getCampaign(id: string)` — select single by id
- `addCampaign(data: NewCampaign)` — insert, return inserted
- `updateCampaign(id: string, data: Partial<NewCampaign>)` — update with updatedAt, return updated
- `deleteCampaign(id: string)` — delete by id
- `getCampaignsByStatus(status: string)` — filter by status

**src/lib/db-segments.ts:**
Functions:
- `loadSegments()` — select all, ordered by createdAt desc
- `getSegment(id: string)` — select single by id
- `addSegment(data: NewSegment)` — insert, return inserted
- `updateSegment(id: string, data: Partial<NewSegment>)` — update with updatedAt, return updated
- `deleteSegment(id: string)` — delete by id (cascade deletes segment_voters)
- `getSegmentVoterIds(segmentId: string)` — select voter ids from segmentVoters junction
- `setSegmentVoters(segmentId: string, voterIds: string[])` — delete existing + insert new junction rows in a transaction
- `updateSegmentCount(segmentId: string)` — count voters in segment and update audienceCount field

Use `import { db } from '@/db'` and `import { voters, campaigns, segments, segmentVoters, ... } from '@/db/schema'` and `import { eq, desc, ilike, sql, and, inArray } from 'drizzle-orm'` as needed.
